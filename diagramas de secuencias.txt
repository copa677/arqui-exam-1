@startuml
title Caso de Uso: Inicio de Sesión (Admin / Encargado)
autonumber

actor "Administrador / Encargado" as User #LightSteelBlue
participant "Frontend (LoginPage)" as FE
participant "API Gateway (:3000)" as GW
participant "MS Usuarios (:3001)" as MS
database "DB Usuarios" as DB
database "LocalStorage" as LS

User -> FE : Ingresa Correo y Contraseña
FE -> FE : Validar formato de datos

FE -> GW : POST /api/login {correo, password}
note right of GW : RUTA PÚBLICA\nNo requiere Token previo

GW -> MS : proxy cURL to :3001/login
MS -> DB : SELECT * FROM usuarios WHERE correo = ?
DB --> MS : Datos (Hash Password, rol_id, info)

MS -> MS : password_verify(password, hash)

alt Autenticación Exitosa
    MS -> MS : Generar JWT (Incluye rol_id)
    MS --> GW : HTTP 201 {token, usuario}
    GW --> FE : HTTP 201 {token, usuario}
    
    FE -> LS : setItem('token', JWT)
    FE -> LS : setItem('user', data)
    
    alt Si rol_id == 1
        FE -> FE : redirect('/admin/usuarios')
        FE --> User : Muestra Panel Administrador
    else Si rol_id == 2
        FE -> FE : redirect('/tecnico')
        FE --> User : Muestra Planilla de Trabajo
    end
    
else Autenticación Fallida
    MS --> GW : HTTP 401 {error: "No autorizado"}
    GW --> FE : HTTP 401 {error}
    FE --> User : Muestra "Credenciales Incorrectas"
end
@enduml

title CU-03: Registro de los Reportadores de Problemas
autonumber

actor "Reportador de Problemas" as Rep #Yellow
participant "Frontend (ReportPage)" as FE
participant "API Gateway (:3000)" as GW
participant "MS Reporte Problemas (:3003)" as MS

Rep -> FE : Ingresa nombres, apellidos, correo y tipo
FE -> FE : Validar campos obligatorios

FE -> GW : POST /api/reportadores {datos_persona}
note right of GW : RUTA PÚBLICA\nCualquier usuario puede reportar

GW -> MS : proxy cURL to :3003/reportadores
MS -> MS : ReportadorController@create()

alt El reporteador ya existe (por Correo)
    MS -> DB : UPDATE reportadores WHERE correo = ?
    DB --> MS : Registro actualizado
else Es un reporteador nuevo
    MS -> DB : INSERT INTO reportadores
    DB --> MS : Registro creado (ID: 5)
end

MS --> GW : HTTP 201 {reportador_id: 5, ...}
GW --> FE : HTTP 201 {reportador_id: 5, ...}

FE -> FE : Guarda ID_Reportador en estado local\npara vincular con la Nota de Problema
FE --> Rep : Habilita paso siguiente (Ubicación/Daños)
@enduml

@startuml
title CU-04: Gestión de Usuarios (CRUD Completo)
autonumber

actor "Administrador" as Admin #LightSteelBlue
participant "Frontend (UsersPage)" as FE
participant "API Gateway (:3000)" as GW
participant "MS Usuarios (:3001)" as MS
database "DB Usuarios" as DB

== Operación: Listar Usuarios ==
Admin -> FE : Accede a Gestión de Usuarios
FE -> GW : GET /api/usuarios
note right of GW : Valida JWT Administrador
GW -> MS : Proxy cURL to :3001
MS -> DB : SELECT * FROM usuarios WHERE activo = true
DB --> MS : Lista de usuarios
MS --> FE : HTTP 200 {usuarios: [...]}
FE --> Admin : Muestra tabla de usuarios

== Operación: Crear Usuario ==
Admin -> FE : Click en "Nuevo" y envía datos
FE -> GW : POST /api/usuarios {datos}
GW -> MS : Proxy cURL to :3001/usuarios
MS -> DB : INSERT INTO usuarios (con hash de clave)
DB --> MS : Creado
MS --> FE : HTTP 201 {usuario}

== Operación: Actualizar Usuario ==
Admin -> FE : Click en "Editar" y guarda cambios
FE -> GW : PUT /api/usuarios/{id} {datos_modificados}
GW -> MS : Proxy cURL to :3001/usuarios/{id}
MS -> DB : UPDATE usuarios SET ... WHERE id = ?
DB --> MS : Actualizado
MS --> FE : HTTP 200 {usuario_editado}

== Operación: Eliminar (Desactivar) Usuario ==
Admin -> FE : Click en "Eliminar"
FE -> GW : DELETE /api/usuarios/{id}
GW -> MS : Proxy cURL to :3001/usuarios/{id}
MS -> DB : UPDATE usuarios SET activo = false WHERE id = ?
DB --> MS : Desactivado
MS --> FE : HTTP 200 {message: "Registro desactivado"}

FE -> FE : Refresca tabla de usuarios
@enduml

title CU-09: Asignación de Problema (Delegación a Técnico)
autonumber

actor "Administrador" as Admin #LightSteelBlue
participant "Frontend (AssignmentsPage)" as FE
participant "API Gateway (:3000)" as GW
participant "MS Gestión (:3004)" as MSG
participant "MS Reporte Problemas (:3003)" as MSR
database "DB Gestión" as DBG
database "DB Reportes" as DBR

Admin -> FE : Selecciona Problema y Técnico
FE -> FE : Prepara datos de asignación

== Paso 1: Crear Vínculo de Asignación ==
FE -> GW : POST /api/asignaciones {detalle_id, usuario_id}
note right of GW : Valida JWT Administrador
GW -> MSG : proxy cURL to :3004/asignaciones
MSG -> DBG : INSERT INTO asignacion (detalle_id, usuario_id, activo=true)
DBG --> MSG : return ID_Asignacion: 50
MSG --> GW : HTTP 201 {id: 50}
GW --> FE : HTTP 201 {id: 50}

== Paso 2: Registrar en Historial ==
FE -> GW : POST /api/historial {asignacion_id: 50, estado: 'Asignado'}
GW -> MSG : proxy cURL to :3004/historial
MSG -> DBG : INSERT INTO historial_estados
DBG --> MSG : OK
MSG --> FE : HTTP 201 OK

== Paso 3: Actualizar Estado del Problema ==
FE -> GW : PUT /api/detalles/{id} {estado_actual: 'Asignado'}
GW -> MSR : proxy cURL to :3003/detalles/{id}
MSR -> DBR : UPDATE detalle_problema SET estado_actual = 'Asignado'
DBR --> MSR : OK
MSR --> FE : HTTP 200 OK

FE --> Admin : Muestra mensaje "Tarea Asignada Exitosamente"
FE -> FE : Filtra el problema de la lista de pendientes
@enduml
